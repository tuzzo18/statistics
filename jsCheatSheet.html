Absolute frequency

                    
function calculateAbsoluteFrequency(list) {
    const frequency = {};
    
    list.forEach((value) => {
        if (frequency[value] !== undefined) {
        frequency[value]++;
        } else {
        frequency[value] = 1;
        }
    });
    
    return frequency;
    }

Generate n random numbers and count their occurrences in k intervals

                    
const myList = [];
const myDictionary = {};

// k different intervals
for (let i = 0; i < k; i++) {
    const key = `[${i / k}; ${(i + 1) / k})`;
    myDictionary[key] = 0;
}

// Generates n random floating-point numbers
for (let j = 1; j <= n; j++) {
    // adds random floating-point numbers >= 0 and < 1
    myList.push(Math.random());
}

// for every generated number, check in which interval it belongs to
for (let y = 0; y < myList.length; y++) {
    for (let i = 0; i < k; i++) {
        const key = `[${i / k}; ${(i + 1) / k})`;
        if (myList[y] >= i / k && myList[y] < (i + 1) / k) {
            myDictionary[key]++;
        }
    }
}
